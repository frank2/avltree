269 class inorder_iterator_base
270 {
271 static_assert(std::is_same<NodeType, SharedNode>::value || std::is_same<NodeType, ConstSharedNode>::value,
272 "Iterator template type must be a SharedNode or a ConstSharedNode.");
273
274 public:
275 using iterator_category = std::forward_iterator_tag;
276 using difference_type = std::ptrdiff_t;
277 using value_type = SharedNode;
278 using pointer = value_type *;
279 using reference = value_type &;
280
281 inorder_iterator_base(NodeType node) : node(node) {
282 if (this->node == nullptr) return;
283
284 while (this->node->_left != nullptr)
285 {
286 this->node = this->node->_left;
287 }
288 }
289 inorder_iterator_base(const inorder_iterator_base &other) : node(other.node) {}
290 ~inorder_iterator_base() {}
291
292 inorder_iterator_base& operator=(const inorder_iterator_base& other) { this->node = other.node; }
293
294 inorder_iterator_base& operator++() {
295 if (this->node == nullptr) { throw exception::NullPointer(); }
296
297 if (this->node->_right != nullptr)
298 {
299 this->node = this->node->_right;
300
301 while (this->node->_left != nullptr)
302 this->node = this->node->_left;
303 }
304 else
305 {
306 auto parent = this->node->_parent;
307
308 while (parent != nullptr && this->node == parent->_right)
309 {
310 this->node = parent;
311 parent = parent->_parent;
312 }
313
314 this->node = parent;
315 }
316
317 return *this;
318 }
319 inorder_iterator_base& operator++(int) { auto tmp = *this; ++(*this); return tmp; }

528 auto rotation_parent = rotation_root->_parent;
529

531 this->_root = pivot_root;
532 else if (rotation_parent->_left == rotation_root)
533 rotation_parent->_left = pivot_root;
534 else if (rotation_parent->_right == rotation_root)
535 rotation_parent->_right = pivot_root;
536

540 left_child->_parent = rotation_root;

542 rotation_root->_right = left_child;

594 virtual void rebalance_node(SharedNode node) {
595 if (node == nullptr) { throw exception::NullPointer(); }
596
597 auto balance = node->balance();
598 if (balance == 0) { return; }
599
600 if (balance < 0)
601 {
602 auto child = node->_left;
603 auto balance = (child != nullptr) ? child->balance() : 0;
604
605 if (balance > 0)
606 {
607 this->rotate_left(child);
608 this->rotate_right(node);
609 }
610 else {
611 this->rotate_right(node);
612 }
613 }
614 else
615 {
616 auto child = node->_right;
617 auto balance = (child != nullptr) ? child->balance() : 0;
618
619 if (balance < 0) {
620 this->rotate_right(child);
621 this->rotate_left(node);
622 }
623 else {
624 this->rotate_left(node);
625 }
626 }

638 virtual void update_node(SharedNode node) {
639 if (node == nullptr) { throw exception::NullPointer(); }
640
641 SharedNode update = node;
642
643 while (update != nullptr)
644 {
645 auto old_height = update->_height;
646 auto new_height = update->new_height();
647 auto balance = update->balance();
648
649 update->_height = new_height;
650
651 if (balance > 1 || balance < -1)
652 {
653 this->rebalance_node(update);
654 return;
655 }
656
657 if (old_height == new_height) { return; }
658
659 if (update->_parent != nullptr)
660 update = update->_parent;
661 }

690 virtual SharedNode add_node(const Value &value) {
691 if (this->_root == nullptr)
692 {
693 this->_root = this->allocate_node(value);
694 ++this->_size;
695 return this->_root;
696 }
697
698 auto key = KeyOfValue()(value);
699 auto traversal = this->search(key);
700 auto last_result = *traversal.rbegin();
701 auto parent = last_result.first;
702 auto branch = last_result.second;
703
704 if (branch == 0) { throw exception::KeyExists(); }
705
706 auto new_node = this->allocate_node(value);
707
708 if (branch < 0) { this->set_left_child(parent, new_node); }
709 else { this->set_right_child(parent, new_node); }
710
711 this->update_node(new_node);
712
713 ++this->_size;
714
715 return new_node;
716 }

727 virtual SharedNode remove_node(const Value &value) {
728 if (this->is_empty()) { throw exception::EmptyTree(); }
729
730 auto key = KeyOfValue()(value);
731 auto traversal = this->search(key);
732 auto last_node = *traversal.rbegin();
733 SharedNode update_node = nullptr;
734
735 if (last_node.second != 0) { throw exception::NodeNotFound(); }
736
737 auto node = last_node.first;
738
739 if (node->is_leaf())
740 {
741 if (node->_parent != nullptr)
742 {
743 if (node->_parent->_left == node)
744 node->_parent->_left.reset();
745 else if (node->_parent->_right == node)
746 node->_parent->_right.reset();
747 }
748
749 if (node == this->_root)
750 {
751 this->_root.reset();
752 }
753 else
754 {
755 auto parent = node->_parent;
756 node->_parent.reset();
757
758 update_node = parent;
759 }
760 }
761 else if (node->_left == nullptr || node->_right == nullptr)
762 {
763 SharedNode replacement_node;
764
765 if (node->_left == nullptr)
766 replacement_node = node->_right;
767 else
768 replacement_node = node->_left;
769
770 replacement_node->_parent = node->_parent;
771
772 if (node == this->_root)
773 {
774 this->_root = replacement_node;
775 update_node = this->_root;
776 }
777 else if (node->_parent->_left == node)
778 {
779 node->_parent->_left = replacement_node;
780 update_node = node->_parent;
781 }
782 else if (node->_parent->_right == node)
783 {
784 node->_parent->_right = replacement_node;
785 update_node = node->_parent;
786 }
787 }
788 else if (node->_left != nullptr && node->_right != nullptr)
789 {
790 auto leftmost = node->_right;
791
792 while (leftmost->_left != nullptr)
793 leftmost = leftmost->_left;
794
795 auto replaced_node = leftmost->_right;
796 auto leftmost_parent = leftmost->_parent;
797
798 if (leftmost_parent == node) {
799 node->_right = replaced_node;
800 }
801 else {
802 leftmost->_parent->_left = replaced_node;
803
804 if (replaced_node != nullptr)
805 replaced_node->_parent = leftmost->_parent;
806 }
807
808 leftmost->copy_node_data(*node);
809
810 if (leftmost->_left != nullptr)
811 leftmost->_left->_parent = leftmost;
812
813 if (leftmost->_right != nullptr)
814 leftmost->_right->_parent = leftmost;
815
816 if (leftmost->_parent != nullptr)
817 {
818 if (node == leftmost->_parent->_left)
819 leftmost->_parent->_left = leftmost;
820 else if (node == leftmost->_parent->_right)
821 leftmost->_parent->_right = leftmost;
822 }
823
824 if (node == this->_root)
825 this->_root = leftmost;
826
827 if (replaced_node != nullptr)
828 update_node = replaced_node;
829 else if (leftmost_parent != node)
830 update_node = leftmost_parent;
831 else
832 update_node = leftmost;
833 }
834
835 node->_left.reset();
836 node->_right.reset();
837 node->_parent.reset();
838
839 --this->_size;
840
841 if (update_node != nullptr) { this->update_node(update_node); }
842
843 return update_node;
844 }
