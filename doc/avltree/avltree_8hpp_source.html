280 inorder_iterator_base(NodeType node) : node(node) {
281 if (this->node == nullptr) return;
282
283 while (this->node->_left != nullptr)
284 {
285 this->node = this->node->_left;

293 inorder_iterator_base& operator++() {
294 if (this->node == nullptr) { throw exception::NullPointer(); }
295
296 if (this->node->_right != nullptr)
297 {
298 this->node = this->node->_right;
299
300 while (this->node->_left != nullptr)
301 this->node = this->node->_left;
302 }
303 else
304 {
305 auto parent = this->node->_parent;
306
307 while (parent != nullptr && this->node == parent->_right)

313 this->node = parent;

593 virtual void rebalance_node(SharedNode node) {
594 if (node == nullptr) { throw exception::NullPointer(); }
595
596 auto balance = node->balance();
597 if (balance == 0) { return; }
598
599 if (balance < 0)
600 {
601 auto child = node->_left;
602 auto balance = (child != nullptr) ? child->balance() : 0;
603
604 if (balance > 0)
605 {
606 this->rotate_left(child);
607 this->rotate_right(node);
608 }
609 else {
610 this->rotate_right(node);
611 }
612 }
613 else
614 {
615 auto child = node->_right;
616 auto balance = (child != nullptr) ? child->balance() : 0;
617
618 if (balance < 0) {
619 this->rotate_right(child);
620 this->rotate_left(node);
621 }
622 else {
623 this->rotate_left(node);
624 }
625 }

637 virtual void update_node(SharedNode node) {
638 if (node == nullptr) { throw exception::NullPointer(); }
639
640 SharedNode update = node;
641
642 while (update != nullptr)
643 {
644 auto old_height = update->_height;
645 auto new_height = update->new_height();
646 auto balance = update->balance();
647
648 update->_height = new_height;
649
650 if (balance > 1 || balance < -1)
651 {
652 this->rebalance_node(update);
653 return;
654 }
655
656 if (old_height == new_height) { return; }
657
658 if (update->_parent != nullptr)
659 update = update->_parent;
660 }

689 virtual SharedNode add_node(const Value &value) {
690 if (this->_root == nullptr)
691 {
692 this->_root = this->allocate_node(value);
693 ++this->_size;
694 return this->_root;
695 }
696
697 auto key = KeyOfValue()(value);
698 auto traversal = this->search(key);
699 auto last_result = *traversal.rbegin();
700 auto parent = last_result.first;
701 auto branch = last_result.second;
702
703 if (branch == 0) { throw exception::KeyExists(); }
704
705 auto new_node = this->allocate_node(value);
706
707 if (branch < 0) { this->set_left_child(parent, new_node); }
708 else { this->set_right_child(parent, new_node); }
709
710 this->update_node(new_node);
711
712 ++this->_size;
713
714 return new_node;
715 }

726 virtual SharedNode remove_node(const Value &value) {
727 if (this->is_empty()) { throw exception::EmptyTree(); }
728
729 auto key = KeyOfValue()(value);
730 auto traversal = this->search(key);
731 auto last_node = *traversal.rbegin();
732 SharedNode update_node = nullptr;
733
734 if (last_node.second != 0) { throw exception::NodeNotFound(); }
735
736 auto node = last_node.first;
737
738 if (node->is_leaf())
739 {
740 if (node->_parent != nullptr)
741 {
742 if (node->_parent->_left == node)
743 node->_parent->_left.reset();
744 else if (node->_parent->_right == node)
745 node->_parent->_right.reset();
746 }
747
748 if (node == this->_root)
749 {
750 this->_root.reset();
751 }
752 else
753 {
754 auto parent = node->_parent;
755 node->_parent.reset();
756
757 update_node = parent;
758 }
759 }
760 else if (node->_left == nullptr || node->_right == nullptr)
761 {
762 SharedNode replacement_node;
763
764 if (node->_left == nullptr)
765 replacement_node = node->_right;
766 else
767 replacement_node = node->_left;
768
769 replacement_node->_parent = node->_parent;
770
771 if (node == this->_root)
772 {
773 this->_root = replacement_node;
774 update_node = this->_root;
775 }
776 else if (node->_parent->_left == node)
777 {
778 node->_parent->_left = replacement_node;
779 update_node = node->_parent;
780 }
781 else if (node->_parent->_right == node)
782 {
783 node->_parent->_right = replacement_node;
784 update_node = node->_parent;
785 }
786 }
787 else if (node->_left != nullptr && node->_right != nullptr)
788 {
789 auto leftmost = node->_right;
790
791 while (leftmost->_left != nullptr)
792 leftmost = leftmost->_left;
793
794 auto replaced_node = leftmost->_right;
795 auto leftmost_parent = leftmost->_parent;
796
797 if (leftmost_parent == node) {
798 node->_right = replaced_node;
799 }
800 else {
801 leftmost->_parent->_left = replaced_node;
802
803 if (replaced_node != nullptr)
804 replaced_node->_parent = leftmost->_parent;
805 }
806
807 leftmost->copy_node_data(*node);
808
809 if (leftmost->_left != nullptr)
810 leftmost->_left->_parent = leftmost;
811
812 if (leftmost->_right != nullptr)
813 leftmost->_right->_parent = leftmost;
814
815 if (leftmost->_parent != nullptr)
816 {
817 if (node == leftmost->_parent->_left)
818 leftmost->_parent->_left = leftmost;
819 else if (node == leftmost->_parent->_right)
820 leftmost->_parent->_right = leftmost;
821 }
822
823 if (node == this->_root)
824 this->_root = leftmost;
825
826 if (replaced_node != nullptr)
827 update_node = replaced_node;
828 else if (leftmost_parent != node)
829 update_node = leftmost_parent;
830 else
831 update_node = leftmost;
832 }
833
834 node->_left.reset();
835 node->_right.reset();
836 node->_parent.reset();
837
838 --this->_size;
839
840 if (update_node != nullptr) { this->update_node(update_node); }
841
842 return update_node;
843 }
